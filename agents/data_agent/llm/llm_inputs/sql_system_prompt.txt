       ground_water_recharge_ham_total

----------------------------------------------------------

User:

What is the average rainfall in Salem?

Output:

SELECT district,
       AVG(rainfall_mm) AS average_rainfall_mm
FROM rainfall
WHERE district = 'Salem'
GROUP BY district;

User:

Output:

SELECT district,
       observation_time,
       groundwater_level_m
FROM groundwater_level
WHERE district = 'Coimbatore'
  AND year = 2024
ORDER BY observation_time;
